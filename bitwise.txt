print("bitwise operator")
a=10
b=4

#AND
print(f"{a} & {b} : {a & b} (binary : {bin(a&b)})")
#OR
print(f"{a} | {b} : {a | b} (binary : {bin(a|b)})")
#XOR
print(f"{a} ^ {b} : {a ^ b} (binary : {bin(a^b)})")
#NOT
print(f"~{a}: {~a} (binary : {bin(~a)})")
#LEFT SHIFT
print(f"{a} <<1  : {a << 1} (binary : {bin(a<<1)})")
#RIGHT SHIFT
print(f"{a} >>1  : {a >> 1} (binary : {bin(a>>1)})")
